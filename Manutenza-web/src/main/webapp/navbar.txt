<script type="text/javascript">

	$(document).ready(function () {
		setRate(4);

		var pathname = window.location.pathname;

		var getUrlParameter = function getUrlParameter(sParam) {
			var sPageURL = decodeURIComponent(window.location.search.substring(1));
			var sURLVariables = sPageURL.split('&');
			var sParameterName;
			var i;

			for (i = 0; i < sURLVariables.length; i++) {
				sParameterName = sURLVariables[i].split('=');

				if (sParameterName[0] === sParam) {
					return sParameterName[1] === undefined ? true : sParameterName[1];
				}
			}
		};

		//pagina home all'avvio dell'app o dopo il login
		if (getUrlParameter('action') === undefined || getUrlParameter('action') === 'logout') {
			var path = pathname + "?action=home";
		} else {
			var path = pathname + "?action=" + getUrlParameter('action');
		}

		//voci di sotto menu
		if ($('.components > li > a[href="' + path + '"]').length === 0) {
                        //voce di menù attiva
                        $('.collapse > li > a[href="' + path + '"]').parent().parent().prev().attr('aria-expanded', 'true');
                        //apri il sottomenù
                        $('.collapse > li > a[href="' + path + '"]').parent().parent().addClass('in');
                        //voce di sottomenù attiva
                        $('.collapse > li > a[href="' + path + '"]').parent().addClass('activeSub');
		}
		//voci della navbar
		else {
			$('.components > li > a[href="' + path + '"]').parent().addClass('active');
		}

		$("#sidebar").mCustomScrollbar({
			theme: "minimal"
		});

		$('#sidebarCollapse').on('click', function () {
			// apre e chiude la navbar
			$('#sidebar, #content').toggleClass('active');
			// close dropdowns
			$('.collapse.in').toggleClass('in');
			// and also adjust aria-expanded attributes we use for the open/closed arrows
			// in our CSS
			$('a[aria-expanded=true]').attr('aria-expanded', 'false');

		});

	});

</script>
